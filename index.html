Greetings. To install the hax.js into your Propane for loads o laughs and hilarity, put this in ~/Library/Application\ Support/Propane/unsupported/caveatPatchor.js:

    var head = $$('head')[0];
    if (head)
    {
        var script = new Element('script', { type: 'text/javascript', src: 'http://chrismo.github.io/propane/hax.js?' + new Date().getTime() });
        head.appendChild(script);
    }
